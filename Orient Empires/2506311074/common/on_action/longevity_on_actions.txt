on_birthday = { 
	trigger = {
		is_landed = yes 
		exists = capital_province

		NOT = { has_character_flag = had_longevity_birthday_action } # 只能触发一次

		trigger_if = {
			limit = {
				any_courtier = {
					count > 0
					has_religion = religion:taoism_religion
				}
			}
			age > 30
		}
		trigger_else_if = {
			limit = {
				has_religion = religion:taoism_religion
			}
			is_adult = yes
		}
		trigger_else = {
			age > 50
		}
		NOT = {
			has_character_flag = had_longevity_0001_event
		}
	}

	events = {
		longevity.0001
	}

	effect = {
		add_character_flag = { # 只能触发一次
			flag = had_longevity_birthday_action
			days = 3650
		}
	}

}
