#设立藩镇 
change_vassal_corps_interaction = {  
	category = interaction_category_vassal
	desc = change_vassal_corps_interaction_desc
	use_diplomatic_range = yes
	common_interaction = yes
	can_send_despite_rejection = yes
	ai_maybe = yes
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			has_culture_group = culture_group:chinese_group
			OR = {
				highest_held_title_tier >= tier_kingdom
				OR = {
					has_government = imperial_government
					is_corps_government_trigger = yes
				}
				AND = {
					is_independent_ruler = yes
					highest_held_title_tier = tier_duchy
				}
			}
		}
		scope:recipient = {
			holds_landed_title = yes
			NOT = {	highest_held_title_tier >= tier_kingdom }
		}
	}
	is_valid = {
		scope:recipient = {
			is_vassal_of = scope:actor
			NOT = { is_corps_government_trigger = yes }
		}
	}
	on_accept = {
		scope:actor = {
			trigger_event = change_vassal_corps_interaction.01
			show_as_tooltip = {
				scope:recipient = {
					add_opinion = {
						modifier = let_me_become_corps_opinion
						target = scope:actor
					}
					change_government_legion_government_effect = yes
				}
			}
		}
	}
	on_decline = {
		scope:actor = {
			trigger_event = change_vassal_corps_interaction.02
			show_as_tooltip = {
				add_opinion = {
					modifier = rejected_my_order_opinion
					target = scope:recipient
				}
				scope:recipient = {
					add_opinion = {
						modifier = want_me_become_corps_opinion
						target = scope:actor
					}
				}
			}
		}
	}
	cooldown_against_recipient = { years = 2 } 
	ai_accept = {
		base = 50
		change_vassal_corps_and_heir_ai_accpet_modifier = yes
	}
	ai_will_do = {
		base = 1
	}
	auto_accept = {
		OR = {
			scope:recipient = {
				is_imprisoned_by = scope:actor
			}
			custom_description = {
				text = "spending_hook"
				subject = scope:actor
				object = scope:recipient
				scope:hook = yes
				scope:actor = { has_strong_hook = scope:recipient }
			}
		}
	}
	send_option = {
		is_valid = {
			scope:actor = {
				has_usable_hook = scope:recipient
			}
		}
		flag = hook
		localization = SCHEME_HOOK
	}
	should_use_extra_icon = {
		scope:actor = { has_usable_hook = scope:recipient }
	}
	extra_icon = "gfx/interface/icons/character_interactions/hook_icon.dds"
	send_options_exclusive = no
}
