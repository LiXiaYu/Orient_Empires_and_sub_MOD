#####显示藩侯
show_chinese_king_effect = {

		if = {
			limit = {
				root = {
					has_government = imperial_government
				}				
			}
			root = {
				## Clear variable always before
				clear_variable_list = $KINGLIST$
				every_vassal = {
					limit = {
						has_title = $KINGTITLE$
					}
					root = {
						add_to_variable_list = {
							name = $KINGLIST$
							target = prev
						}
					}					
				}
			}
		}
		if = {
			limit = {
				root = {
					NOT = {
						has_government = imperial_government
					}	
					any_liege_or_above = {
						has_government = imperial_government
					}				
				}
			}
			root = { clear_variable_list = $KINGLIST$ }
			root.liege = {
				every_vassal = {
					limit = {
						has_title = $KINGTITLE$
					}
					root = {
						add_to_variable_list = {
							name = $KINGLIST$
							target = prev
						}
					}					
				}
			}
		}	
}
#####创建藩王头衔
OE_create_kingdom_effect = {
		play_music_cue_once = { TRACK = mx_cue_ck3_vocal_maintheme }
		scope:actor = {
			create_title_and_vassal_change = {
				type = created
				save_scope_as = change
				add_claim_on_loss = no
			}
			$KINGTITLE$ = {
				change_title_holder = {
				holder = scope:secondary_actor
				change = scope:change
				}	
			}
			resolve_title_and_vassal_change = scope:change
		}
}
