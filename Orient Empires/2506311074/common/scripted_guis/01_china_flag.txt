show_chinesename_gui = {
	scope = character
	is_shown = {
		has_culture_group = culture_group:chinese_group
	}
}
#####非中文名
not_chinesename_gui = {
	scope = character
	is_shown = {
		NOT = { has_culture_group = culture_group:chinese_group}
		has_dynasty = yes
	}
}
