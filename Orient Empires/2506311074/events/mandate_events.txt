mandate.0004 = {
	type = character_event
	title = mandate.0004.title
	desc = {	
		desc = mandate.0004.desc.normal
	}
	theme = prison
	left_portrait = root
	right_portrait = {
		character = scope:defender
		animation = fear
	}
	trigger = {
		root = { has_primary_title = title:e_celestia }
	}
	immediate = {
		
	}
	option = {
		name = mandate.0004.a
		remove_character_flag = fired_from_event
		if = {
			limit = {
				scope:defender = { is_alive = yes }
			}
			execute_prisoner_effect = {
				VICTIM = scope:defender
				EXECUTIONER = root
			}
		}
		else = {
			show_as_tooltip = {
				scope:defender = {
					death = {
						death_reason = death_execution
						killer = root
					}
				}
			}
		}
	}
	option = {
		name = mandate.0004.b
		remove_character_flag = fired_from_event
		if = {
			limit = {
				scope:defender = { is_alive = yes }
			}
			hard_imprison_character_effect = {
				TARGET = scope:defender
				IMPRISONER = root
			}
		}
	}
}

	}
	option = {
		name = mandate.0005.a

		ai_potential = {
			always = yes
		}

		scope:attacker = {
		create_title_and_vassal_change = {
			type = conquest
			save_scope_as = change
			add_claim_on_loss = no
		}
			every_held_title = {
				if = {
					limit = {
						OR = { 
							target_is_de_jure_liege_or_above = title:k_yan
							target_is_de_jure_liege_or_above = title:d_yunzhong
						}						
					}
					change_title_holder = {
						holder = title:e_cathay.holder
						change = scope:change
					}
				}
			}
			every_vassal_or_below = {
				if = {
					limit = {
						any_held_title = {
							OR = {
								target_is_de_jure_liege_or_above = title:k_yan
								target_is_de_jure_liege_or_above = title:d_yunzhong
							}
						}
					}
					change_liege = {
						liege = title:e_cathay.holder
						change = scope:change
					}
				}
			}
		resolve_title_and_vassal_change = scope:change
		}
		hidden_effect = {
			title:k_yan = { de_jure_liege = e_cathay }
			title:k_khitan = { de_jure_liege = e_cathay }
			title:k_jurchen = { de_jure_liege = e_cathay }
			title:d_yunzhong = { de_jure_liege = k_khitan }
			title:c_linzhou_shaan.holder = { 
				becomes_independent = yes
			}
		}
		ai_chance = 10000
	}

	option = {
		name = mandate.0005.b
