# longevity 
namespace = longevity

longevity.0001 = {
	type = character_event

	title = longevity.0001.t
	desc = longevity.0001.desc
	theme = physical_health

	left_portrait = root

	trigger = {
		is_immortal = no
		NOT = { has_character_flag = had_longevity_0001_event }
		is_ai = no # debug
	}

	immediate = {
		add_character_flag = {
			flag = had_longevity_0001_event
			days = 3650
		}
	}

	option = {
		name = longevity.0001.a
		if = {
			limit = {
				is_immortal = no
			}

			trigger_event = longevity.0002
		}
	}
	option = {
		ai_chance = {
			base = 100
		}
		name = longevity.0001.b
	}
}

longevity.0002 = {
	type = character_event

	title = longevity.0002.t
	desc = longevity.0002.desc
	theme = physical_health

	left_portrait = {
		character = root
	}

	option = {
		name = longevity.0002.a

		add_trait = longevity_trait
		set_immortal_age = this.age
	}

}
